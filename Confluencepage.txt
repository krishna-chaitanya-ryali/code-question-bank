def parse_cascade_date(
    date_str
):
    if not date_str:
        return datetime.datetime.min

    try:
        # Remove timezone like IST/BST/EST
        date_without_tz = (
            date_str.rsplit(
                " ",
                1
            )[0]
        )

        return datetime.datetime.strptime(
            date_without_tz,
            "%d %b %Y %I:%M:%S %p"
        )

    except Exception:
        return datetime.datetime.min
