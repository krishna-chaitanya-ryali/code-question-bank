SELECT
    ACTION_ID,
    STATUS,
    CASCADE_VERSION
FROM (
    SELECT
        ACTION_ID,
        STATUS,
        CASCADE_VERSION
    FROM METRIC_CASCADE_ACTIONS
    WHERE METRIC_DETAIL_ID = :1
      AND CHILD_RMM_ID = :2
      AND ACTION_TYPE = :3
    ORDER BY CASCADE_VERSION DESC,
             ACTION_ID DESC
)
WHERE ROWNUM = 1
