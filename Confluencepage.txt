CREATE TABLE METRIC_CASCADE_ACTIONS (
    ACTION_ID        NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    METRIC_DETAIL_ID NUMBER NOT NULL,
    PARENT_RMM_ID    NUMBER NOT NULL,
    ACTION_TYPE      VARCHAR2(20), -- 'CASCADE' / 'REMINDER'
    ACTION_BY        VARCHAR2(50),
    ACTION_DT        TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT FK_MCA_METRIC
    FOREIGN KEY (METRIC_DETAIL_ID)
    REFERENCES RAP_METRICS_DETAILS(METRIC_DETAIL_ID)
);


INSERT INTO METRIC_CASCADE_ACTIONS (
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    ACTION_TYPE,
    ACTION_BY
)
VALUES (
    :metric_detail_id,
    :rmm_id,
    'CASCADE',
    :user_id
);

SELECT 
    RMDT.METRIC_DETAIL_ID,
    RMDT.METRICS_DISPLAY AS METRIC_NAME,
    RMDT.METRIC_NAME_CASCADE,
    RMDT.THRESHOLD_CASCADE,
    RMDT.METRIC_APPLICABILITY,

    MCA.CASCADE_DT,
    MCA.ACTION_BY

FROM MEET_INSTC MI

JOIN RAP 
    ON MI.MEET_INSTC_ID = RAP.RAP_INSTANCE_ID

JOIN RAP_METRICS_DETAILS RMDT 
    ON RAP.RAP_ID = RMDT.RAP_ID

/* 🔹 Get latest cascade action (date + user) */
LEFT JOIN (
    SELECT 
        METRIC_DETAIL_ID,
        MAX(ACTION_DT) AS CASCADE_DT,
        MAX(ACTION_BY) KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS ACTION_BY
    FROM METRIC_CASCADE_ACTIONS
    WHERE ACTION_TYPE = 'CASCADE'
    GROUP BY METRIC_DETAIL_ID
) MCA
    ON MCA.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID

WHERE MI.RMM_ID = :rmm_id
AND MI.MEET_INSTC_ID = :meet_instc_id
AND RMDT.METRIC_FLAG <> 'removed'

/* 🔹 Only metrics that are cascaded */
AND EXISTS (
    SELECT 1
    FROM METRIC_CASCADE_LOCATIONS MCL
    WHERE MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
)


def build_metric_assignment_response(vRecs):
    response = []

    for r in vRecs:
        response.append({
            "metric_detail_id": r.get("METRIC_DETAIL_ID"),
            "metric_name": r.get("METRIC_NAME"),
            "metric_name_cascade": r.get("METRIC_NAME_CASCADE"),
            "threshold_cascade": r.get("THRESHOLD_CASCADE"),
            "metric_applicability": r.get("METRIC_APPLICABILITY"),

            "cascade_date": r.get("CASCADE_DT"),
            "actioned_by": r.get("ACTION_BY")
        })

    return response
