    SELECT 
        METRIC_DETAIL_ID,
        LISTAGG(RMM_ID, ',') WITHIN GROUP (ORDER BY RMM_ID) AS CASCADED_RMM_IDS
    FROM RAPUSER.METRIC_CASCADE_LOCATIONS
    GROUP BY METRIC_DETAIL_ID
) MCL_AGG
ON MCL_AGG.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID

    data["cascaded_to_child_metrics"] = [int(x) for x in rmm_ids.split(",")]
else:
    data["cascaded_to_child_metrics"] = []
