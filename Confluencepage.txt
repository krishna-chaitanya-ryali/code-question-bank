924SELECT
    MCA.ACTION_ID,
    MCA.METRIC_DETAIL_ID,
    MCA.TARGET_METRIC_DETAIL_ID,
    MCA.STATUS,
    MCA.CASCADE_VERSION,
    MCA.ACTION_BY,
    MCA.CHILD_RMM_ID,

    -- METRIC TITLE
    CASE
        WHEN RMDT.METRICS_DISPLAY IS NULL
        THEN RMPM.METRICS_DISP
        ELSE RMDT.METRICS_DISPLAY
    END AS METRIC_TITLE,

    -- MASTER METRIC ID
    RMDT.MASTER_METRIC_ID,

    -- RMM NAME
    RRG.RMM_NAME,

    -- RISK TYPE
    RRT.RISK_TYPE,

    -- USER NAME
    CASE
        WHEN MCA.ACTION_BY = 'SYSTEM'
        THEN 'SYSTEM'
        ELSE UT.USER_NAME
    END AS ACTIONED_BY

FROM METRIC_CASCADE_ACTIONS MCA

LEFT JOIN RAP_METRICS_DETAILS RMDT
    ON RMDT.METRIC_DETAIL_ID = MCA.METRIC_DETAIL_ID

LEFT JOIN RAP_METRICS_PACK_MAPPING RMPM
    ON RMPM.RAP_METRICS_MAPPING_ID = RMDT.RAP_METRICS_MAPPING_ID

LEFT JOIN RAP RP
    ON RP.RAP_ID = RMDT.RAP_ID

LEFT JOIN RAP_RISK_TYPE RRT
    ON RRT.RAP_RISK_TYPE_ID = RP.RAP_RISK_TYPE_ID

LEFT JOIN RAP_RMM_GROUP RRG
    ON RRG.RMM_ID = MCA.CHILD_RMM_ID

LEFT JOIN USER_TAB UT
    ON UT.EMP_ID = MCA.ACTION_BY

WHERE
(
    MCA.METRIC_DETAIL_ID = :1
    OR MCA.TARGET_METRIC_DETAIL_ID = :2
)

ORDER BY MCA.ACTION_ID DESC
