SELECT
    METRIC_DETAIL_ID,

    MAX(ACTION_DT) AS CASCADE_DT,

    MAX(ACTION_BY) KEEP (
        DENSE_RANK LAST ORDER BY ACTION_DT
    ) AS ACTION_BY,

    MAX(ACTION_DT_STR) KEEP (
        DENSE_RANK LAST ORDER BY ACTION_DT
    ) AS CASCADE_DT_STR

FROM METRIC_CASCADE_ACTIONS
WHERE ACTION_TYPE = 'CASCADE'
GROUP BY METRIC_DETAIL_ID
