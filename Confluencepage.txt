SELECT DISTINCT r.LINEMANAGER_ID
FROM RAPUSER.RECERTIFICATION_EMAIL_LIST r
WHERE r.MAIL_SENT_COUNT = 1
  AND r.LINEMANAGER_ID IS NOT NULL
  AND r.SYSTEM_REVOKE_DONE_IND = 'N'
  AND NOT EXISTS (
        SELECT 1
        FROM RAPUSER.USER_TAB ut
        WHERE ut.LINEMANAGER_ID = r.LINEMANAGER_ID
          AND ut.CREAT_DT >= DATE '2026-03-16'
  );
