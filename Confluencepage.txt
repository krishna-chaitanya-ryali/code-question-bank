SELECT 
    RMDT.RAP_METRICS_MAPPING_ID,
    MCL.RMM_ID,

    -- ✅ Last Modified Date
    RMDT.UPDT_DT AS LAST_MODIFIED_DATE,

    -- ✅ Last Modified By (EMP_ID priority + PSID fallback)
    COALESCE(
        UT_UPD.USER_NAME,
        (
            SELECT UT2.USER_NAME
            FROM USER_TAB UT2
            WHERE UT2.PSID = RMDT.UPDT_BY
            FETCH FIRST 1 ROW ONLY
        )
    ) AS LAST_MODIFIED_BY,

    -- ✅ Existing Functional Approver (DON’T TOUCH)
    UT.USER_NAME AS FUNCTION_APPROVER_NAME

FROM RAP_METRICS_DETAILS RMDT

JOIN RAP_METRICS_PACK_MAPPING RMPM 
    ON RMDT.RAP_METRICS_MAPPING_ID = RMPM.RAP_METRICS_MAPPING_ID

LEFT JOIN METRIC_CASCADE_LOCATIONS MCL 
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID

-- ✅ Existing join (keep as is)
LEFT JOIN USER_TAB UT 
    ON UT.USER_ID = RMDT.FUNCTION_APPROVER

-- ✅ Primary join (EMP_ID)
LEFT JOIN USER_TAB UT_UPD 
    ON UT_UPD.EMP_ID = RMDT.UPDT_BY

WHERE RMDT.METRIC_DETAIL_ID = :metric_id
AND RMDT.METRIC_FLAG <> 'removed';
