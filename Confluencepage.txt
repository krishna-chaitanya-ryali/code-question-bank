SELECT
    L.METRIC_DETAIL_ID,
    L.PARENT_RMM_ID AS LOC_PARENT,
    L.CHILD_RMM_ID AS LOC_CHILD,
    L.CASCADE_VERSION AS LOC_VERSION,

    A.PARENT_RMM_ID AS ACT_PARENT,
    A.CHILD_RMM_ID AS ACT_CHILD,
    A.CASCADE_VERSION AS ACT_VERSION,

    A.ACTION_BY,
    A.ACTION_DT_STR

FROM METRIC_CASCADE_LOCATIONS L

LEFT JOIN METRIC_CASCADE_ACTIONS A
ON A.METRIC_DETAIL_ID = L.METRIC_DETAIL_ID
AND A.CHILD_RMM_ID = L.CHILD_RMM_ID
AND A.CASCADE_VERSION = L.CASCADE_VERSION
AND A.ACTION_TYPE='REMOVE'

WHERE L.METRIC_DETAIL_ID = 237357
ORDER BY L.CHILD_RMM_ID;
