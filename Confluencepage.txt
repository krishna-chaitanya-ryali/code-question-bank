SELECT
    METRIC_DETAIL_ID,
    CHILD_RMM_ID,
    ACTION_TYPE,
    CASCADE_VERSION
FROM METRIC_CASCADE_LOCATIONS
WHERE METRIC_DETAIL_ID = 237335
ORDER BY CHILD_RMM_ID, ACTION_TYPE;


SELECT
    METRIC_DETAIL_ID,
    CHILD_RMM_ID,
    ACTION_TYPE,
    CASCADE_VERSION
FROM METRIC_CASCADE_ACTIONS
WHERE METRIC_DETAIL_ID = 237335
ORDER BY CHILD_RMM_ID, ACTION_TYPE;
