@views.route("/cascade_get_metric_details", methods=["GET"])
def cascade_get_metric_details():
    try:
        metric_id = reqs.args.get('METRIC_DETAIL_ID')
        details_list = sql_getData(522, [int(metric_id)])

        if not details_list:
            return jsonify([])   # ✅ Safe return

        base_record = details_list[0]

        # 🔹 Collect cascaded locations
        rmm_ids = list({row['RMM_ID'] for row in details_list if row.get('RMM_ID')})
        base_record['CASCADED_LOCATIONS'] = rmm_ids

        # 🔹 FORMAT DATE
        if base_record.get("LAST_MODIFIED_DATE"):
            base_record["LAST_MODIFIED_DATE"] = format_datetime(base_record["LAST_MODIFIED_DATE"])

        # 🔹 Existing logic (safe access)
        if base_record.get("DT_ID") and base_record["DT_ID"] > GRAF_CUTOFF_DATE:
            base_record["RAS_CATEGORY"] = 0
            if base_record.get("RAP_METRICS_CATEGORY") is not None:
                base_record["RAS_CATEGORY"] = base_record["RAP_METRICS_CATEGORY"]

        return jsonify([base_record])

    except Exception as e:
        log_error.error('Error | mod_views | cascade_get_metric_details | ' + str(e))
        return ("Error in getting metric_details")
