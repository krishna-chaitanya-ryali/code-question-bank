SELECT
    METRIC_DETAIL_ID,
    TARGET_METRIC_DETAIL_ID,
    STATUS,
    ACTION_TYPE
FROM METRIC_CASCADE_ACTIONS
WHERE TARGET_METRIC_DETAIL_ID = 236737;

SELECT *
FROM METRIC_CASCADE_REVIEW_STATE
WHERE CASCADED_METRIC_DETAIL_ID = 236737;



SELECT
    TARGET_METRIC_DETAIL_ID,
    STATUS,
    ACTION_DT
FROM METRIC_CASCADE_ACTIONS
WHERE TARGET_METRIC_DETAIL_ID = 236737
ORDER BY ACTION_DT DESC;
