def submit_cascade_review(vInfo):

    try:

        cascaded_metric_detail_id = vInfo.get(
            "cascaded_metric_detail_id"
        )

        accepted_fields = (
            vInfo.get("acceptedFieldKeys") or []
        )

        rejected_fields = (
            vInfo.get("rejectedFieldKeys") or []
        )

        remarks = vInfo.get(
            "remarks",
            ""
        )

        dispensation_files = (
            vInfo.get("dispensationFiles") or []
        )

        vUser = str(myinfo())

        file_name = None
        file_path = None

        # file metadata
        if dispensation_files:

            uploaded_file = dispensation_files[0]

            file_name = uploaded_file.get("name")

            file_path = (
                f"/nas/metric_cascade/"
                f"{cascaded_metric_detail_id}/"
                f"{file_name}"
            )

        # calculate status
        status = get_cascade_review_status(
            accepted_fields,
            rejected_fields
        )

        # store review rows
        insert_review_details(
            cascaded_metric_detail_id,
            accepted_fields,
            rejected_fields,
            remarks,
            file_name,
            file_path,
            vUser
        )

        # update main status
        sql_insertData(
            391,
            "UPDATE_STATUS",
            [
                status,
                vUser,
                cascaded_metric_detail_id
            ]
        )

        return jsonify({
            "status": status
        })

    except Exception as e:

        log_error.error(
            f"submit_cascade_review : {str(e)}"
        )

        return jsonify("Error")
