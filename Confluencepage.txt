LEFT JOIN (
    SELECT
        METRIC_DETAIL_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION,
        MAX(ACTION_DT) AS CASCADE_DT,
        MAX(ACTION_BY)
            KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS ACTION_BY,
        MAX(ACTION_DT_STR)
            KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS CASCADE_DT_STR
    FROM METRIC_CASCADE_ACTIONS
    WHERE ACTION_TYPE = 'CASCADE'
    GROUP BY
        METRIC_DETAIL_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION
) MCA
ON MCA.METRIC_DETAIL_ID = MCL.METRIC_DETAIL_ID
AND MCA.CASCADE_VERSION = MCL.CASCADE_VERSION
AND MCA.CHILD_RMM_ID = MCL.RMM_ID
