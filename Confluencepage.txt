COALESCE(
    UT.USER_NAME,
    (
        SELECT UT2.USER_NAME
        FROM USER_TAB UT2
        WHERE UT2.PSID = MCA.ACTION_BY
        FETCH FIRST 1 ROW ONLY
    )
) AS ACTION_BY


CASE
    WHEN MCA.ACTION_BY = 'SYSTEM'
    THEN 'SYSTEM'

    ELSE COALESCE(
        UT.USER_NAME,
        (
            SELECT UT2.USER_NAME
            FROM USER_TAB UT2
            WHERE UT2.PSID = MCA.ACTION_BY
            FETCH FIRST 1 ROW ONLY
        )
    )
END AS ACTION_BY
