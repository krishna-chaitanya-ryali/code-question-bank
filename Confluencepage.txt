def derive_cascade_request_status(
    accepted_count,
    rejected_count,
    total_fields=9
):

    reviewed_count = (
        accepted_count +
        rejected_count
    )

    # -------------------------------------
    # NOTHING REVIEWED
    # -------------------------------------

    if reviewed_count == 0:

        return "NOT_STARTED"

    # -------------------------------------
    # PARTIAL REVIEW
    # -------------------------------------

    if reviewed_count < total_fields:

        return "PARTIAL_REVIEW"

    # -------------------------------------
    # ALL FIELDS REVIEWED
    # -------------------------------------

    return "FULLY_REVIEWED"


def derive_next_action(status):

    if not status:

        return "REVIEW_POPUP"

    status = status.strip()

    if status == "Accepted":

        return "EDIT_POPUP"

    if status == "Partial Review":

        return "DISPENSATION_POPUP"

    if status == "Rejected":

        return "DISPENSATION_POPUP"

    return "REVIEW_POPUP"

def get_cascade_review_status(
    accepted_fields,
    rejected_fields,
    total_fields=9,
    is_draft=False
):

    if is_draft:
        return "In Progress"

    accepted_count = len(accepted_fields)

    rejected_count = len(rejected_fields)

    decided_count = (
        accepted_count +
        rejected_count
    )

    # nothing selected
    if decided_count == 0:
        return "Not Started"

    # partial selection
    if decided_count < total_fields:
        return "In Progress"

    # all rejected
    if rejected_count == total_fields:
        return "Rejected"

    # mix accept + reject
    if rejected_count > 0:
        return "Partial Review"

    # all accepted
    return "Accepted"


def insert_dispensation_details(
    cascaded_metric_detail_id,
    rejected_fields,
    remarks,
    file_name,
    file_path,
    user
):

    rejected_fields = rejected_fields or []

    for field in rejected_fields:

        sql_insertData(
            393,
            "DISPENSATION_ID",
            [
                cascaded_metric_detail_id,
                field,
                remarks,
                file_name,
                file_path,
                user
            ]
        )


def submit_cascade_review(vInfo):

    try:

        cascaded_metric_detail_id = (
            vInfo.get(
                "cascaded_metric_detail_id"
            )
        )

        accepted_fields = (
            vInfo.get(
                "acceptedFieldKeys"
            ) or []
        )

        rejected_fields = (
            vInfo.get(
                "rejectedFieldKeys"
            ) or []
        )

        vUser = str(myinfo())

        status = get_cascade_review_status(
            accepted_fields,
            rejected_fields
        )

        # save review selections
        save_review_state(
            cascaded_metric_detail_id,
            accepted_fields,
            rejected_fields,
            vUser
        )

        # update workflow status
        sql_insertData(
            391,
            "NA",
            [
                status,
                vUser,
                cascaded_metric_detail_id
            ]
        )

        return jsonify({
            "status": status
        })

    except Exception as e:

        log_error.error(
            f"submit_cascade_review : {str(e)}"
        )

        return jsonify("Error")


def submit_dispensation_log(vInfo):

            vInfo.get(
                "dispensationFiles"
            ) or []
        )

        file_name = None

        # file upload metadata
        if dispensation_files:

            )

            )

            file_path = (
                FOLDER_EXCEL +
                f"{cascaded_metric_detail_id}/" +
                f"{file_name}"
            )

        insert_dispensation_details(
            cascaded_metric_detail_id,

            file_path,
            vUser
        )

        return jsonify({
            "status": "Success"

    except Exception as e:

        log_error.error(
            f"submit_dispensation_log : {str(e)}"
        )

        return jsonify("Error")
