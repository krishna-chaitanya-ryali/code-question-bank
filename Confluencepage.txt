
SELECT 
    ut.LINEMANAGER_ID,
    ut.LM_NAME,
    ut.LM_MAIL,
    COUNT(*) AS RESPONSE_COUNT
FROM USER_TAB ut
JOIN RECERT_RESPONSE rr 
    ON rr.REPORTEES_ID = ut.EMP_ID
WHERE rr.ACCESS_STATUS IN ('Maintain', 'Revoke')
AND ut.LINEMANAGER_ID IS NOT NULL
GROUP BY 
    ut.LINEMANAGER_ID,
    ut.LM_NAME,
    ut.LM_MAIL;


SELECT 
    ut.LINEMANAGER_ID,
    ut.LM_NAME,
    ut.LM_MAIL,
    COUNT(DISTINCT ut.EMP_ID) AS TOTAL_REPORTEES,
    COUNT(DISTINCT rr.REPORTEES_ID) AS RESPONDED_REPORTEES
FROM USER_TAB ut
LEFT JOIN RECERT_RESPONSE rr 
    ON rr.REPORTEES_ID = ut.EMP_ID
    AND rr.ACCESS_STATUS IN ('Maintain', 'Revoke')
WHERE ut.LINEMANAGER_ID IS NOT NULL
GROUP BY 
    ut.LINEMANAGER_ID,
    ut.LM_NAME,
    ut.LM_MAIL;
