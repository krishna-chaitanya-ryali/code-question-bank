SELECT
    FIELD_KEY,
    REVIEW_TYPE
FROM METRIC_CASCADE_REVIEW_STATE
WHERE CASCADED_METRIC_DETAIL_ID = :1
394

def get_cascade_review_state(vInfo):

    try:

        cascaded_metric_detail_id = (
            vInfo.get(
                "cascaded_metric_detail_id"
            )
        )

        review_rows = sql_getData(
            394,
            [
                cascaded_metric_detail_id
            ]
        )

        accepted_keys = []
        rejected_keys = []

        if review_rows and review_rows != -1:

            for row in review_rows:

                review_type = (
                    row.get("REVIEW_TYPE")
                )

                field_key = (
                    row.get("FIELD_KEY")
                )

                if review_type == "Accepted":

                    accepted_keys.append(
                        field_key
                    )

                elif review_type == "Rejected":

                    rejected_keys.append(
                        field_key
                    )

        return jsonify({

            "acceptedFieldKeys":
                accepted_keys,

            "rejectedFieldKeys":
                rejected_keys
        })

    except Exception as e:

        log_error.error(
            "get_cascade_review_state : "
            + str(e)
        )

        return jsonify({
            "status": "Error"
        })
