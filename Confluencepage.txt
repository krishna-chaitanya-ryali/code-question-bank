JOIN METRIC_CASCADE_LOCATIONS MCL 
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID

JOIN METRIC_CASCADE_LOCATIONS MCL 
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
    AND EXISTS (
        SELECT 1
        FROM METRIC_CASCADE_ACTIONS MCA2
        WHERE MCA2.METRIC_DETAIL_ID = MCL.METRIC_DETAIL_ID
        AND MCA2.CHILD_RMM_ID = MCL.RMM_ID
        AND MCA2.CASCADE_VERSION = MCL.CASCADE_VERSION
        AND MCA2.ACTION_TYPE = 'CASCADE'
        AND MCA2.IS_ACTIVE = 'Y'
    )
