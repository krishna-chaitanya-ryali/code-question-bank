CREATE TABLE RAPUSER01.METRIC_CASCADE_LOCATIONS (
    CASCADED_ID NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,

    METRIC_DETAIL_ID NUMBER NOT NULL,
    RMM_ID NUMBER NOT NULL,
    LOCATION_TYPE VARCHAR2(50 BYTE) NOT NULL,

    CREAT_DT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CREAT_BY VARCHAR2(50 BYTE),

    UPDT_DT TIMESTAMP,
    UPDT_BY VARCHAR2(50 BYTE),

    CONSTRAINT UK_MCL UNIQUE (METRIC_DETAIL_ID, RMM_ID, LOCATION_TYPE),

    CONSTRAINT FK_MCL_METRIC_DETAIL
        FOREIGN KEY (METRIC_DETAIL_ID)
        REFERENCES RAPUSER01.RAP_METRICS_DETAILS (METRIC_DETAIL_ID)
        ON DELETE CASCADE
);


CREATE INDEX IDX_MCL_METRIC_DETAIL_ID
ON RAPUSER01.METRIC_CASCADE_LOCATIONS (METRIC_DETAIL_ID);
