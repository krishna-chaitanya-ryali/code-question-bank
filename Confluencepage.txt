SELECT
    MCA.ACTION_ID,
    MCA.METRIC_DETAIL_ID,
    MCA.TARGET_METRIC_DETAIL_ID,
    MCA.STATUS,
    MCA.CASCADE_VERSION,
    MCA.ACTION_BY,
    MCA.CHILD_RMM_ID,
    MCA.PARENT_RMM_ID,
    MCA.ACTION_DT_STR,

    -- METRIC TITLE
    CASE
        WHEN RMDT.METRICS_DISPLAY IS NULL
        THEN RMPM.METRICS_DISP
        ELSE RMDT.METRICS_DISPLAY
    END AS METRIC_TITLE,

    -- MASTER METRIC ID
    RMDT.MASTER_METRIC_ID,

    -- METRIC CATEGORY
    RMDT.RAP_METRIC_CATEGORY,

    -- RISK TYPE
    RRT.RISK_TYPE_DISP,

    -- CASCADE LOCATION
    CASE

        -- CURRENT SCREEN IS PARENT
        WHEN MCA.PARENT_RMM_ID = :2
        THEN CHILD_RRG.RMM_DISPLAY_NAME

        -- CURRENT SCREEN IS CHILD
        WHEN MCA.CHILD_RMM_ID = :3
        THEN PARENT_RRG.RMM_DISPLAY_NAME

        -- FALLBACK
        ELSE CHILD_RRG.RMM_DISPLAY_NAME

    END AS CASCADE_LOCATION,

    -- ACTIONED BY
    CASE

        WHEN MCA.ACTION_BY = 'SYSTEM'
        THEN 'SYSTEM'

        WHEN MCA.STATUS IN (
            'Awaiting Acceptance'
        )
        THEN NULL

        ELSE COALESCE(
            UT.USER_NAME,
            (
                SELECT UT2.USER_NAME
                FROM USER_TAB UT2
                WHERE UT2.PSID = MCA.ACTION_BY
                FETCH FIRST 1 ROW ONLY
            )
        )

    END AS ACTIONED_BY,

    -- DISPENSATION
    CASE
        WHEN MCDD.CASCADE_METRIC_DETAIL_ID IS NOT NULL
        THEN 'Y'
        ELSE 'N'
    END AS DISPENSATION,

    -- REMARKS AVAILABLE
    CASE
        WHEN MCDD.REMARKS IS NOT NULL
        THEN 'Y'
        ELSE 'N'
    END AS HAS_REMARKS,

    -- REMARKS
    MCDD.REMARKS

FROM METRIC_CASCADE_ACTIONS MCA

LEFT JOIN RAP_METRICS_DETAILS RMDT
    ON RMDT.METRIC_DETAIL_ID = MCA.METRIC_DETAIL_ID

LEFT JOIN RAP_METRICS_PACK_MAPPING RMPM
    ON RMPM.RAP_METRICS_MAPPING_ID = RMDT.RAP_METRICS_MAPPING_ID

LEFT JOIN RAP RP
    ON RP.RAP_ID = RMDT.RAP_ID

LEFT JOIN RAP_RISK_TYPE RRT
    ON RRT.RISK_TYPE_ID = RP.RISK_TYPE_ID

-- CHILD LOCATION
LEFT JOIN RAP_RMM_GROUP CHILD_RRG
    ON CHILD_RRG.RMM_ID = MCA.CHILD_RMM_ID

-- PARENT LOCATION
LEFT JOIN RAP_RMM_GROUP PARENT_RRG
    ON PARENT_RRG.RMM_ID = MCA.PARENT_RMM_ID

LEFT JOIN USER_TAB UT
    ON UT.EMP_ID = MCA.ACTION_BY

LEFT JOIN METRIC_CASCADE_DISPENSATION_DETAILS MCDD
    ON MCDD.CASCADE_METRIC_DETAIL_ID = MCA.METRIC_DETAIL_ID
    AND MCDD.RMM_ID = MCA.CHILD_RMM_ID

WHERE
    MCA.METRIC_DETAIL_ID = :1

AND MCA.ACTION_TYPE = 'CASCADE'

ORDER BY
    MCA.ACTION_ID DESC
