FROM (
    SELECT
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        MAX(CHILD_RMM_ID) AS CHILD_RMM_ID,
        CASCADE_VERSION,
        MAX(CASCADE_STATUS) AS CASCADE_STATUS,
        MIN(IS_ACTIVE) AS IS_ACTIVE,
        MAX(CREAT_DT) AS CREAT_DT,

        ROW_NUMBER() OVER (
            PARTITION BY
                METRIC_DETAIL_ID,
                CASE
                    WHEN PARENT_RMM_ID = :5 THEN 'CURRENT_LOCATION'
                    WHEN MAX(CHILD_RMM_ID) = :6 THEN 'PARENT_LOCATION'
                END
            ORDER BY
                CASCADE_VERSION DESC,
                MAX(CREAT_DT) DESC
        ) AS RN

    FROM METRIC_CASCADE_LOCATIONS

    WHERE ACTION_TYPE = 'REMOVE'
      AND (
            PARENT_RMM_ID = :7
            OR CHILD_RMM_ID = :8
      )

    GROUP BY
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CASCADE_VERSION
) MCL
