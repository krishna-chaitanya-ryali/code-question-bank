def get_cascade_review_status(
    accepted_fields,
    rejected_fields,
    total_fields=9,
    is_draft=False
):

    if is_draft:
        return "In Progress"

    accepted_count = len(accepted_fields)
    rejected_count = len(rejected_fields)

    decided_count = accepted_count + rejected_count

    # not all reviewed
    if decided_count < total_fields:
        return "In Progress"

    # all rejected
    if rejected_count == total_fields:
        return "Rejected"

    # finalized
    return "Accepted"

import os
from werkzeug.utils import secure_filename

NAS_DIRECTORY = "/nas/metric_cascade/"


def save_review_file(file_obj, cascaded_metric_detail_id):

    folder_path = os.path.join(
        NAS_DIRECTORY,
        str(cascaded_metric_detail_id)
    )

    os.makedirs(folder_path, exist_ok=True)

    filename = secure_filename(file_obj.filename)

    full_path = os.path.join(folder_path, filename)

    file_obj.save(full_path)

    return filename, full_path


def insert_review_details(
    cascaded_metric_detail_id,
    accepted_fields,
    rejected_fields,
    remarks,
    file_name,
    file_path,
    user
):

    # clear old draft rows
    sql_insertData(
        389,
        "DELETE_REVIEW_ROWS",
        [cascaded_metric_detail_id]
    )

    # insert accepted fields
    for field in accepted_fields:

        sql_insertData(
            390,
            "INSERT_REVIEW_ROW",
            [
                cascaded_metric_detail_id,
                field,
                "Accepted",
                remarks,
                file_name,
                file_path,
                user
            ]
        )

    # insert rejected fields
    for field in rejected_fields:

        sql_insertData(
            390,
            "INSERT_REVIEW_ROW",
            [
                cascaded_metric_detail_id,
                field,
                "Rejected",
                remarks,
                file_name,
                file_path,
                user
            ]
        )

    try:

        vInfo = reqs.get_json()

        cascaded_metric_detail_id = vInfo.get(
            "cascaded_metric_detail_id"
        )

        data = sql_getData(
            392,
            [cascaded_metric_detail_id]
        )

        accepted_fields = []
        rejected_fields = []

        remarks = ""
        file_name = ""
        file_path = ""

        for row in data:

            if row["REVIEW_TYPE"] == "Accepted":
                accepted_fields.append(
                    row["FIELD_KEY"]
                )

            elif row["REVIEW_TYPE"] == "Rejected":
                rejected_fields.append(
                    row["FIELD_KEY"]
                )

            remarks = row["REMARKS"]
            file_name = row["FILE_NAME"]
            file_path = row["FILE_PATH"]

        return jsonify({
            "acceptedFieldKeys": accepted_fields,
            "rejectedFieldKeys": rejected_fields,
            "remarks": remarks,
            "fileName": file_name,
            "filePath": file_path
        })

    except Exception as e:
        log_error.error(str(e))
        return jsonify("Error")
