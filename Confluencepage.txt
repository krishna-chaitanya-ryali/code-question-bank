CREATE TABLE METRIC_CASCADE_LOCATIONS (
    CASCADED_ID BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,

    METRIC_DETAIL_ID BIGINT NOT NULL,

    RMM_ID BIGINT NOT NULL,

    LOCATION_TYPE VARCHAR(20) NOT NULL,  
    -- GROUP / REGION / GB / MARKET

    CREAT_DT TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    CREAT_BY VARCHAR(50),

    UPDT_DT TIMESTAMP,
    UPDT_BY VARCHAR(50),

    CONSTRAINT FK_MCL_METRIC_DETAIL
        FOREIGN KEY (METRIC_DETAIL_ID)
        REFERENCES RAP_METRIC_DETAILS (METRIC_ID)
        ON DELETE CASCADE
);


ALTER TABLE METRIC_CASCADE_LOCATIONS
ADD CONSTRAINT UK_MCL_UNIQUE
UNIQUE (METRIC_DETAIL_ID, RMM_ID, LOCATION_TYPE);


CREATE INDEX IDX_MCL_METRIC_DETAIL_ID
ON METRIC_CASCADE_LOCATIONS (METRIC_DETAIL_ID);
