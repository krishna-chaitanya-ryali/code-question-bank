SELECT
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE
FROM METRIC_CASCADE_LOCATIONS
WHERE METRIC_DETAIL_ID = 237357
ORDER BY ACTION_TYPE, CHILD_RMM_ID, CASCADE_VERSION;



SELECT
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE,
    IS_ACTIVE,
    ROW_NUMBER() OVER (
        PARTITION BY
            METRIC_DETAIL_ID,
            CASE
                WHEN PARENT_RMM_ID = 14 THEN 'CURRENT_LOCATION'
                WHEN CHILD_RMM_ID = 14 THEN 'PARENT_LOCATION'
            END
        ORDER BY CHILD_RMM_ID
    ) RN
FROM METRIC_CASCADE_LOCATIONS
WHERE
    ACTION_TYPE = 'REMOVE'
AND (
    PARENT_RMM_ID = 14
    OR CHILD_RMM_ID = 14
)
AND METRIC_DETAIL_ID = 237357;
