JOIN (
    SELECT
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,

        MIN(CHILD_RMM_ID) AS CHILD_RMM_ID,

        MAX(CASCADE_VERSION) AS CASCADE_VERSION,

        MAX(CASCADE_STATUS) AS CASCADE_STATUS,

        MAX(IS_ACTIVE) AS IS_ACTIVE

    FROM METRIC_CASCADE_LOCATIONS

    WHERE ACTION_TYPE = 'REMOVE'

    GROUP BY
        METRIC_DETAIL_ID,
        PARENT_RMM_ID

) MCL
