SELECT 
    RMDT.METRIC_DETAIL_ID,
    RMDT.METRICS_DISPLAY AS METRIC_NAME,
    RMDT.METRIC_NAME_CASCADE,
    RMDT.THRESHOLD_CASCADE,
    RMDT.METRIC_APPLICABILITY,
    MCA.CASCADE_DT,
    MCA.ACTION_BY

FROM MEET_INSTC MI
JOIN RAP ON MI.MEET_INSTC_ID = RAP.RAP_INSTANCE_ID
JOIN RAP_METRICS_DETAILS RMDT ON RAP.RAP_ID = RMDT.RAP_ID

LEFT JOIN (
    SELECT 
        METRIC_DETAIL_ID,
        MAX(ACTION_DT) AS CASCADE_DT,
        MAX(ACTION_BY) KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS ACTION_BY
    FROM METRIC_CASCADE_ACTIONS
    WHERE ACTION_TYPE = 'CASCADE'
    GROUP BY METRIC_DETAIL_ID
) MCA ON MCA.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID

WHERE MI.RMM_ID = :current_rmm_id
AND MI.MEET_INSTC_ID = :meet_instc_id
AND RMDT.METRIC_FLAG <> 'removed'

AND EXISTS (
    SELECT 1
    FROM METRIC_CASCADE_LOCATIONS MCL
    WHERE MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
);

def build_response(child_data, parent_data):

    def format_data(records):
        return [
            {
                "metric_detail_id": r.get("METRIC_DETAIL_ID"),
                "metric_name": r.get("METRIC_NAME"),
                "metric_name_cascade": r.get("METRIC_NAME_CASCADE"),
                "threshold_cascade": r.get("THRESHOLD_CASCADE"),
                "metric_applicability": r.get("METRIC_APPLICABILITY"),
                "cascade_date": r.get("CASCADE_DT"),
                "actioned_by": r.get("ACTION_BY")
            }
            for r in records
        ]

    return {
        "cascade_to_child": format_data(child_data),
        "cascade_from_parent": format_data(parent_data)
    }

def cascade_metrics_action(vInfo):
    try:
        rmm_id = vInfo.get("rmm_id")
        instance_id = vInfo.get("instanceid")
        user_id = vInfo.get("user_id")
        metric_ids = vInfo.get("metric_detail_ids", [])

        if not metric_ids:
            return jsonify({"status": "error", "message": "No metrics selected"})

        if not rmm_id or not instance_id or not user_id:
            return jsonify({"status": "error", "message": "Missing required fields"})

        for metric_id in metric_ids:
            sql_execute(
                """
                INSERT INTO METRIC_CASCADE_ACTIONS (
                    METRIC_DETAIL_ID,
                    PARENT_RMM_ID,
                    ACTION_TYPE,
                    ACTION_BY,
                    ACTION_DT
                ) VALUES (
                    :1, :2, 'CASCADE', :3, CURRENT_TIMESTAMP
                )
                """,
                [metric_id, rmm_id, user_id]
            )

        db.commit()

        return jsonify({
            "status": "success",
            "message": "Metrics cascaded successfully"
        })

    except Exception as e:
        db.rollback()
        log_error.error(f"Error in cascade_metrics_action: {str(e)}")
        return jsonify({
            "status": "error",
            "message": "Failed to cascade metrics"
        })
