COALESCE(
    RMDT.UPDT_BY,
    RMDT.CREAT_BY,
    RMPM.UPDT_BY,
    RMPM.CREAT_BY
) AS LAST_MODIFIED_PSID,

COALESCE(
    (
        SELECT UT2.USER_NAME
        FROM USER_TAB UT2
        WHERE UT2.PSID = COALESCE(
            RMDT.UPDT_BY,
            RMDT.CREAT_BY,
            RMPM.UPDT_BY,
            RMPM.CREAT_BY
        )
        FETCH FIRST 1 ROW ONLY
    ),
    (
        SELECT UT3.USER_NAME
        FROM USER_TAB UT3
        WHERE UT3.EMP_ID = COALESCE(
            RMDT.UPDT_BY,
            RMDT.CREAT_BY,
            RMPM.UPDT_BY,
            RMPM.CREAT_BY
        )
        FETCH FIRST 1 ROW ONLY
    )
) AS LAST_MODIFIED_BY
