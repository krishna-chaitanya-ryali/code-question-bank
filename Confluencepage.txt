SELECT
    cols.column_name
FROM
    all_constraints cons,
    all_cons_columns cols
WHERE
    cons.constraint_name = cols.constraint_name
    AND cons.owner = cols.owner
    AND cons.constraint_type = 'U'
    AND cons.constraint_name = 'UK_MCL';



ALTER TABLE METRIC_CASCADE_LOCATIONS
DROP CONSTRAINT UK_MCL;



ALTER TABLE METRIC_CASCADE_LOCATIONS
ADD CONSTRAINT UK_MCL
UNIQUE
(
    METRIC_DETAIL_ID,
    RMM_ID,
    CASCADE_VERSION
);
