LEFT JOIN (
    SELECT
        *
    FROM (
        SELECT
            MCDD.*,
            ROW_NUMBER() OVER (
                PARTITION BY MCDD.CASCADED_METRIC_DETAIL_ID
                ORDER BY MCDD.CREAT_DT DESC
            ) AS RN
        FROM METRIC_CASCADE_DISPENSATION_DETAILS MCDD
    )
    WHERE RN = 1
) MCDD
    ON MCDD.CASCADED_METRIC_DETAIL_ID = MCA.METRIC_DETAIL_ID



SELECT
    CASCADED_METRIC_DETAIL_ID,
    COUNT(*)
FROM METRIC_CASCADE_DISPENSATION_DETAILS
GROUP BY CASCADED_METRIC_DETAIL_ID
HAVING COUNT(*) > 1;
