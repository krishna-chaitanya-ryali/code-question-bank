def derive_next_action(
    cascade_request_status
):

    if not cascade_request_status:
        return "REVIEW_POPUP"

    cascade_request_status = (
        cascade_request_status.strip()
    )

    if cascade_request_status in [
        "NOT_STARTED",
        "PARTIAL_REVIEW"
    ]:
        return "REVIEW_POPUP"

    return ""
