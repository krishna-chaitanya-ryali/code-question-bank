SELECT
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE,
    IS_ACTIVE,

    ROW_NUMBER() OVER (
        PARTITION BY
            METRIC_DETAIL_ID,
            CASE
                WHEN PARENT_RMM_ID = 14 THEN 'CURRENT_LOCATION'
                WHEN CHILD_RMM_ID = 14 THEN 'PARENT_LOCATION'
            END
        ORDER BY CHILD_RMM_ID
    ) RN

FROM METRIC_CASCADE_LOCATIONS
WHERE ACTION_TYPE = 'REMOVE'
AND (
        PARENT_RMM_ID = 14
     OR CHILD_RMM_ID = 14
)
AND METRIC_DETAIL_ID = 237357

ORDER BY CHILD_RMM_ID;



SELECT

    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,

    MAX(ACTION_DT) CASCADE_DT,

    MAX(ACTION_BY)
        KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) ACTION_BY,

    MAX(ACTION_DT_STR)
        KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) ACTION_DT_STR,

    MAX(IS_ACTIVE)
        KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) IS_ACTIVE

FROM METRIC_CASCADE_ACTIONS

WHERE ACTION_TYPE='REMOVE'
AND METRIC_DETAIL_ID=237357

GROUP BY

    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION

ORDER BY
    CHILD_RMM_ID,
    CASCADE_VERSION;


WITH MCL AS
(
    SELECT
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION,
        ACTION_TYPE,
        IS_ACTIVE,

        ROW_NUMBER() OVER (
            PARTITION BY
                METRIC_DETAIL_ID,
                CASE
                    WHEN PARENT_RMM_ID = 14 THEN 'CURRENT_LOCATION'
                    WHEN CHILD_RMM_ID = 14 THEN 'PARENT_LOCATION'
                END
            ORDER BY CHILD_RMM_ID
        ) RN

    FROM METRIC_CASCADE_LOCATIONS

    WHERE ACTION_TYPE='REMOVE'
    AND (
            PARENT_RMM_ID=14
         OR CHILD_RMM_ID=14
    )
),

MCA AS
(
    SELECT

        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION,

        MAX(ACTION_DT_STR)
            KEEP(
                DENSE_RANK LAST
                ORDER BY ACTION_DT
            ) ACTION_DT_STR,

        MAX(ACTION_BY)
            KEEP(
                DENSE_RANK LAST
                ORDER BY ACTION_DT
            ) ACTION_BY

    FROM METRIC_CASCADE_ACTIONS

    WHERE ACTION_TYPE='REMOVE'

    GROUP BY

        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION
)

SELECT

    MCL.METRIC_DETAIL_ID,

    MCL.PARENT_RMM_ID LOC_PARENT,
    MCL.CHILD_RMM_ID LOC_CHILD,
    MCL.CASCADE_VERSION LOC_VERSION,

    MCA.PARENT_RMM_ID ACT_PARENT,
    MCA.CHILD_RMM_ID ACT_CHILD,
    MCA.CASCADE_VERSION ACT_VERSION,

    MCA.ACTION_BY,
    MCA.ACTION_DT_STR,

    MCL.RN

FROM MCL

LEFT JOIN MCA

ON MCA.METRIC_DETAIL_ID=MCL.METRIC_DETAIL_ID
AND MCA.PARENT_RMM_ID=MCL.PARENT_RMM_ID
AND MCA.CHILD_RMM_ID=MCL.CHILD_RMM_ID
AND MCA.CASCADE_VERSION=MCL.CASCADE_VERSION

WHERE MCL.METRIC_DETAIL_ID=237357

ORDER BY
    MCL.CHILD_RMM_ID,
    MCL.CASCADE_VERSION;
