SELECT 
    RMDT.METRIC_DETAIL_ID,
    RMDT.METRICS_DISPLAY AS METRIC_NAME,
    RMDT.METRIC_NAME_CASCADE,
    RMDT.THRESHOLD_CASCADE,
    RMDT.METRIC_APPLICABILITY
FROM RAP_METRICS_DETAILS RMDT
WHERE RMDT.RMM_ID = :rmm_id
  AND RMDT.METRIC_FLAG <> 'removed'
  AND EXISTS (
        SELECT 1 
        FROM METRIC_CASCADE_LOCATIONS MCL
        WHERE MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
  )


LEFT JOIN (
    SELECT 
        METRIC_DETAIL_ID,
        LISTAGG(RMM_ID, ',') WITHIN GROUP (ORDER BY RMM_ID) AS CASCADED_RMMS
    FROM METRIC_CASCADE_LOCATIONS
    GROUP BY METRIC_DETAIL_ID
) MCL_AGG
ON RMDT.METRIC_DETAIL_ID = MCL_AGG.METRIC_DETAIL_ID



for record in result:
    rmm_ids = record.get("CASCADED_RMMS")

    record["cascaded_locations"] = (
        [int(x) for x in rmm_ids.split(",") if x.strip()]
        if rmm_ids else []
    )
