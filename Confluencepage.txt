JOIN (
    SELECT
        METRIC_DETAIL_ID,
        RMM_ID,
        LOCATION_TYPE,
        MAX(CASCADE_VERSION) AS CASCADE_VERSION
    FROM METRIC_CASCADE_LOCATIONS
    WHERE IS_ACTIVE = 'Y'
    GROUP BY
        METRIC_DETAIL_ID,
        RMM_ID,
        LOCATION_TYPE
) MCL
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
