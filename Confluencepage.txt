def get_cascade_review_status(accepted_fields, rejected_fields, total_fields=9):

    accepted_count = len(accepted_fields)
    rejected_count = len(rejected_fields)

    if accepted_count == total_fields:
        return "Accepted"

    elif rejected_count == total_fields:
        return "Rejected"

    elif accepted_count > 0 and rejected_count > 0:
        return "In Progress"

    elif rejected_count > 0:
        return "Rejected"

    return "Awaiting Acceptance"



NAS_BASE_PATH = "/nas/metric_cascade/"


import os
from werkzeug.utils import secure_filename

def save_dispensation_file(file_obj, cascaded_metric_detail_id):

    folder_path = os.path.join(
        NAS_BASE_PATH,
        str(cascaded_metric_detail_id)
    )

    os.makedirs(folder_path, exist_ok=True)

    filename = secure_filename(file_obj.filename)

    full_path = os.path.join(folder_path, filename)

    file_obj.save(full_path)

    return filename, full_path



def insert_review_details(
    cascaded_metric_detail_id,
    accepted_fields,
    rejected_fields,
    remarks,
    file_name,
    file_path,
    user
):

    dbCon = rapdb()

    sql = """
        INSERT INTO METRIC_CASCADE_REVIEW_DETAILS (
            CASCADED_METRIC_DETAIL_ID,
            FIELD_KEY,
            REVIEW_STATUS,
            REMARKS,
            FILE_NAME,
            FILE_PATH,
            CREATED_BY,
            CREATED_DT
        )
        VALUES (
            :1, :2, :3, :4, :5, :6, :7, CURRENT_TIMESTAMP
        )
    """

    values = []

    for field in accepted_fields:
        values.append((
            cascaded_metric_detail_id,
            field,
            "Accepted",
            remarks,
            file_name,
            file_path,
            user
        ))

    for field in rejected_fields:
        values.append((
            cascaded_metric_detail_id,
            field,
            "Rejected",
            remarks,
            file_name,
            file_path,
            user
        ))

    dbCon.bulk_execute(sql, values)

def cascade_review_submit():

        vInfo = reqs.get_json()

        cascaded_metric_detail_id = vInfo.get(

        )

        )

            "rejectedFieldKeys",

        )

        remarks = vInfo.get(
            "remarks",
            ""
        )

        uploaded_file = request.files.get("file")

        file_path = None

        if uploaded_file:
            file_name, file_path = save_dispensation_file(

            rejected_fields,
            remarks,

            file_path,
            vUser
        )

                cascaded_metric_detail_id

        )

    except Exception as e:
        log_error.error(str(e))
