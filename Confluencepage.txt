    METRIC_DETAIL_ID,
    RMM_ID,
    CASCADE_VERSION,
    CASCADE_STATUS,
    IS_ACTIVE
FROM METRIC_CASCADE_LOCATIONS
WHERE METRIC_DETAIL_ID = 234214;

    METRIC_DETAIL_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE,
    IS_ACTIVE
