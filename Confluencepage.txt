SELECT
    TARGET_METRIC_DETAIL_ID
FROM (
    SELECT
        TARGET_METRIC_DETAIL_ID
    FROM METRIC_CASCADE_ACTIONS
    WHERE
        METRIC_DETAIL_ID = :1
        AND CHILD_RMM_ID = :2
        AND STATUS = 'Accepted'
    ORDER BY
        CASCADE_VERSION DESC,
        ACTION_ID DESC
)
WHERE ROWNUM = 1
