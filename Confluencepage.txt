def get_allowed_children(rmm_id):
    parent_level = get_rmm_level(rmm_id)

    # ❌ Stop at level 2
    if parent_level is None or parent_level >= 2:
        return set()

    # 🔥 SPECIAL CASE: GLOBAL (LEVEL 0)
    if parent_level == 0:
        rows = sql_getData("GET_LEVEL1_RMMS", [])
        return set(r["RMM_ID"] for r in rows)

    # ✅ Normal case (GB → LOB, Region → Market)
    rows = sql_getData("GET_VALID_CHILDREN", [rmm_id])
    return set(r["RMM_ID"] for r in rows)


-- GET_LEVEL1_RMMS
SELECT RMM_ID
FROM RAP_RMM_GROUP
WHERE RMM_LEVEL = 1
AND ACTIVE = 'Y'
