LEFT JOIN (

    SELECT
        MCA1.METRIC_DETAIL_ID,

        MCA1.CHILD_RMM_ID,

        MCA1.CASCADE_VERSION,

        MCA1.ACTION_BY,

        MCA1.ACTION_DT_STR AS CASCADE_DT_STR,

        MCA1.STATUS,

        MCA1.ACTION_ID

    FROM METRIC_CASCADE_ACTIONS MCA1

    WHERE MCA1.ACTION_TYPE = 'CASCADE'

) MCA
    ON MCA.METRIC_DETAIL_ID = MCL.METRIC_DETAIL_ID

   AND MCA.CASCADE_VERSION = MCL.CASCADE_VERSION

   AND MCA.CHILD_RMM_ID = MCL.CHILD_RMM_ID
