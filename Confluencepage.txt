@app.route(
    "/cascade_status_log",
    methods=["GET"]
)
def cascade_status_log():

    try:

        metric_detail_id = request.args.get(
            "metric_detail_id"
        )

        rmm_id = request.args.get(
            "rmmId"
        )

        result = sql_getData(
            924,
            [
                metric_detail_id,
                rmm_id
            ]
        )

        if not result:
            return jsonify({
                "status": "error",
                "message": "No data found"
            })

        data = []

        for row in result:

            data.append({

                "Metric Title": row.get(
                    "METRIC_TITLE"
                ),

                "Metric ID": row.get(
                    "MASTER_METRIC_ID"
                ),

                "Risk Type": row.get(
                    "RISK_TYPE_DISP"
                ),

                "Metric Category": row.get(
                    "RAP_METRICS_CATEGORY"
                ),

                "Cascade Location": row.get(
                    "CASCADE_LOCATION"
                ),

                "Cascade Status": row.get(
                    "STATUS"
                ),

                "Dispensation": row.get(
                    "DISPENSATION"
                ),

                "Go Live Date": row.get(
                    "GO_LIVE_DT"
                ),

                "Accepted / Rejected By": row.get(
                    "ACTIONED_BY"
                ),

                "Accepted / Rejected Date": row.get(
                    "ACTION_DT_STR"
                ),

                "Remarks": row.get(
                    "REMARKS"
                )

            })

        df = pd.DataFrame(data)

        file_name = (
            f"Cascade_Status_Log_{metric_detail_id}.xlsx"
        )

        file_path = os.path.join(
            app.config["UPLOAD_FOLDER"],
            file_name
        )

        df.to_excel(
            file_path,
            index=False
        )

        return send_file(
            file_path,
            as_attachment=True
        )

    except Exception as e:

        log_error.error(
            f"Error in cascade_status_log: {str(e)}"
        )

        return jsonify({
            "status": "error",
            "message": "Failed to download report"
        })
