WHERE
(
    PARENT_RMM_ID = :7
    OR CHILD_RMM_ID = :8
)
AND
(
    PARENT_RMM_ID = :7

    OR EXISTS
    (
        SELECT 1
        FROM METRIC_CASCADE_ACTIONS MCA
        WHERE MCA.METRIC_DETAIL_ID =
              METRIC_CASCADE_LOCATIONS.METRIC_DETAIL_ID

          AND MCA.CHILD_RMM_ID =
              METRIC_CASCADE_LOCATIONS.CHILD_RMM_ID

          AND MCA.CASCADE_VERSION =
              METRIC_CASCADE_LOCATIONS.CASCADE_VERSION

          AND MCA.ACTION_TYPE='REMOVE'

          AND MCA.IS_ACTIVE='Y'
    )
)
