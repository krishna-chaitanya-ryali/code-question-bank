ROW_NUMBER() OVER (
    PARTITION BY
        METRIC_DETAIL_ID,
        CASE
            WHEN PARENT_RMM_ID = :5 THEN 'CURRENT_LOCATION'
            WHEN CHILD_RMM_ID = :6 THEN 'PARENT_LOCATION'
        END
    ORDER BY
        CASE
            WHEN EXISTS (
                SELECT 1
                FROM METRIC_CASCADE_ACTIONS MCA
                WHERE MCA.METRIC_DETAIL_ID = METRIC_CASCADE_LOCATIONS.METRIC_DETAIL_ID
                  AND MCA.CHILD_RMM_ID = METRIC_CASCADE_LOCATIONS.CHILD_RMM_ID
                  AND MCA.CASCADE_VERSION = METRIC_CASCADE_LOCATIONS.CASCADE_VERSION
                  AND MCA.ACTION_TYPE = 'REMOVE'
            )
            THEN 0
            ELSE 1
        END,
        CHILD_RMM_ID
) AS RN


SELECT
    METRIC_DETAIL_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE
FROM METRIC_CASCADE_LOCATIONS
WHERE METRIC_DETAIL_ID = 237357
ORDER BY CHILD_RMM_ID;
