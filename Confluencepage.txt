JOIN METRIC_CASCADE_LOCATIONS MCL
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID


JOIN (
    SELECT
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        ACTION_TYPE,
        CASCADE_VERSION,

        MAX(CHILD_RMM_ID) AS CHILD_RMM_ID,
        MAX(CASCADE_STATUS) AS CASCADE_STATUS,
        MAX(IS_ACTIVE) AS IS_ACTIVE

    FROM METRIC_CASCADE_LOCATIONS

    WHERE ACTION_TYPE = 'REMOVE'

    GROUP BY
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        ACTION_TYPE,
        CASCADE_VERSION
) MCL
    ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
