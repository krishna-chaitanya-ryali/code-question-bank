  AND PARENT_RMM_ID = :4
  AND CHILD_RMM_ID = :5

  AND CASCADE_VERSION = :6;

    VISIBLE_AT_RMM_ID = :1,

    UPDT_DT = CURRENT_TIMESTAMP,
