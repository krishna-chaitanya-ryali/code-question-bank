SELECT RMDT.RAP_METRICS_MAPPING_ID,
       MCL.CHILD_RMM_ID,
       RMDT.UPDT_DT AS LAST_MODIFIED_DATE,
       RMDT.UPDT_BY AS LAST_MODIFIED_PSID,
       COALESCE(
           UT_UPD.USER_NAME,
           (
               SELECT UT2.USER_NAME
               FROM USER_TAB UT2
               WHERE UT2.PSID = RMDT.UPDT_BY
               FETCH FIRST 1 ROW ONLY
           )
       ) AS LAST_MODIFIED_BY
FROM
RAP_METRICS_DETAILS RMDT
JOIN RAP_METRICS_PACK_MAPPING RMPM ON RMDT.RAP_METRICS_MAPPING_ID=RMPM.RAP_METRICS_MAPPING_ID
JOIN RAP R ON RMDT.RAP_ID=R.RAP_ID
JOIN RAP_RISK_TYPE RRT ON R.RISK_TYPE_ID=RRT.RISK_TYPE_ID
LEFT JOIN RAP_METRICS_DETAILS PRMDT ON PRMDT.METRIC_DETAIL_ID = RMDT.PARENT_METRIC_ID
LEFT JOIN RAP_METRICS_PACK_MAPPING PRMPM ON PRMPM.RAP_METRICS_MAPPING_ID= PRMDT.RAP_METRICS_MAPPING_ID
JOIN MEET_INSTC RI ON RI.MEET_INSTC_ID = R.RAP_INSTANCE_ID
LEFT JOIN USER_TAB UT ON UT.USER_ID=RMDT.FUNCTION_APPROVER
LEFT JOIN USER_TAB UT_UPD ON UT_UPD.EMP_ID=RMDT.UPDT_BY
LEFT JOIN METRIC_CASCADE_LOCATIONS MCL ON MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
WHERE
RMDT.METRIC_DETAIL_ID=:1
AND RMDT.METRIC_FLAG <> 'removed'
