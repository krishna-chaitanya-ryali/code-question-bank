SELECT
    COLUMN_NAME,
    DATA_TYPE
FROM USER_TAB_COLUMNS
WHERE TABLE_NAME = 'METRIC_CASCADE_ACTIONS'
AND COLUMN_NAME IN (
    'ACTION_DT',
    'ACTION_DT_STR',
    'LAST_UPDATED_DT',
    'UPDT_DT'
);


SELECT
    ACTION_DT,
    LAST_UPDATED_DT,
    ACTION_DT_STR,
    STATUS
FROM METRIC_CASCADE_ACTIONS
WHERE CHILD_RMM_ID = :1
ORDER BY ACTION_DT DESC;



SELECT
    METRIC_DETAIL_ID,
    ACTION_DT,
    LAST_UPDATED_DT,
    NVL(
        LAST_UPDATED_DT,
        ACTION_DT
    ) AS SORT_DATE
FROM METRIC_CASCADE_ACTIONS
WHERE CHILD_RMM_ID = :1
ORDER BY
    NVL(
        LAST_UPDATED_DT,
        ACTION_DT
    ) DESC;
