SELECT
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE,
    STATUS,
    IS_ACTIVE
FROM METRIC_CASCADE_ACTIONS
WHERE METRIC_DETAIL_ID = <working_metric_detail_id>
ORDER BY ACTION_DT;



SELECT
    METRIC_DETAIL_ID,
    PARENT_RMM_ID,
    CHILD_RMM_ID,
    CASCADE_VERSION,
    ACTION_TYPE,
    IS_ACTIVE
FROM METRIC_CASCADE_LOCATIONS
WHERE METRIC_DETAIL_ID = <working_metric_detail_id>
ORDER BY CHILD_RMM_ID;
