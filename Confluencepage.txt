def insert_review_details(
    cascaded_metric_detail_id,
    accepted_fields,
    rejected_fields,
    remarks,
    file_name,
    file_path,
    user
):

    accepted_fields = accepted_fields or []
    rejected_fields = rejected_fields or []

    # clear old rows
    sql_insertData(
        389,
        "DELETE_REVIEW_ROWS",
        [cascaded_metric_detail_id]
    )

    # insert accepted fields
    for field in accepted_fields:

        sql_insertData(
            390,
            "INSERT_REVIEW_ROW",
            [
                cascaded_metric_detail_id,
                field,
                "Accepted",
                remarks,
                file_name,
                file_path,
                user
            ]
        )

    # insert rejected fields
    for field in rejected_fields:

        sql_insertData(
            390,
            "INSERT_REVIEW_ROW",
            [
                cascaded_metric_detail_id,
                field,
                "Rejected",
                remarks,
                file_name,
                file_path,
                user
            ]
        )
