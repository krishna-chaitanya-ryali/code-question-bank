SELECT 
    RMDT.METRIC_DETAIL_ID,

    CASE 
        WHEN RMDT.METRICS_DISPLAY IS NULL 
        THEN RMPM.METRICS_DISP 
        ELSE RMDT.METRICS_DISPLAY 
    END AS METRIC_NAME,

    RMDT.METRIC_NAME_CASCADE,
    RMDT.THRESHOLD_CASCADE,
    RMDT.METRIC_APPLICABILITY,

    MCA.CASCADE_DT,
    MCA.ACTION_BY,
    NVL(MCA.STATUS, 'AWAITING_ACCEPTANCE') AS STATUS

FROM MEET_INSTC MI

JOIN RAP 
    ON MI.MEET_INSTC_ID = RAP.RAP_INSTANCE_ID

JOIN RAP_METRICS_DETAILS RMDT 
    ON RAP.RAP_ID = RMDT.RAP_ID

JOIN RAP_METRICS_PACK_MAPPING RMPM
    ON RMDT.RAP_METRICS_MAPPING_ID = RMPM.RAP_METRICS_MAPPING_ID

-- 🔥 latest cascade action (moved ABOVE WHERE)
LEFT JOIN (
    SELECT 
        METRIC_DETAIL_ID,
        CHILD_RMM_ID,
        MAX(ACTION_DT) AS CASCADE_DT,
        MAX(ACTION_BY) KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS ACTION_BY,
        MAX(STATUS) KEEP (DENSE_RANK LAST ORDER BY ACTION_DT) AS STATUS
    FROM METRIC_CASCADE_ACTIONS
    WHERE ACTION_TYPE = 'CASCADE'
    GROUP BY METRIC_DETAIL_ID, CHILD_RMM_ID
) MCA
    ON MCA.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
    AND MCA.CHILD_RMM_ID = :rmm_id

WHERE MI.RMM_ID = :rmm_id
AND MI.MEET_INSTC_ID = :meet_instc_id
AND RMDT.METRIC_FLAG <> 'removed'

-- 🔥 ensures only cascaded metrics for this child
AND EXISTS (
    SELECT 1
    FROM METRIC_CASCADE_LOCATIONS MCL
    WHERE MCL.METRIC_DETAIL_ID = RMDT.METRIC_DETAIL_ID
    AND MCL.RMM_ID = :rmm_id
)

ORDER BY METRIC_NAME;
