def derive_cascade_request_status(
    accepted_count,
    rejected_count,
    total_fields=9
):

    reviewed_count = (
        accepted_count + rejected_count
    )

    # -------------------------------------
    # NOTHING REVIEWED
    # -------------------------------------

    if reviewed_count == 0:

        return "NOT_STARTED"

    # -------------------------------------
    # PARTIAL REVIEW
    # -------------------------------------

    if reviewed_count < total_fields:

        return "PARTIAL_REVIEW"

    # -------------------------------------
    # ALL FIELDS REVIEWED
    # -------------------------------------

    return "FULLY_REVIEWED"
