MAX(CHILD_RMM_ID) KEEP (
    DENSE_RANK LAST
    ORDER BY CASCADE_VERSION
) AS CHILD_RMM_ID,

MAX(CASCADE_STATUS) KEEP (
    DENSE_RANK LAST
    ORDER BY CASCADE_VERSION
) AS CASCADE_STATUS,

MAX(IS_ACTIVE) KEEP (
    DENSE_RANK LAST
    ORDER BY CASCADE_VERSION
) AS IS_ACTIVE


MAX(CHILD_RMM_ID) AS CHILD_RMM_ID,
MAX(CASCADE_STATUS) AS CASCADE_STATUS,
MAX(IS_ACTIVE) AS IS_ACTIVE
