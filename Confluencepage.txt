CASE
    WHEN MCL.PARENT_RMM_ID = :2
    THEN 'CURRENT_LOCATION'

    WHEN MCL.CHILD_RMM_ID = :2
    THEN 'PARENT_LOCATION'

    ELSE NULL
END AS REMOVE_SOURCE_TYPE


"REMOVE_SOURCE_TYPE": record.get(
    "REMOVE_SOURCE_TYPE"
),

WHERE (
        MCL.PARENT_RMM_ID = :2
        OR
        MCL.CHILD_RMM_ID = :2
      )
  AND MI.MEET_INSTC_ID = :3
  AND RMDT.METRIC_FLAG = 'remove


"REMOVE_SOURCE_TYPE": record.get(
    "REMOVE_SOURCE_TYPE"
),



SELECT
    RMDT.MASTER_METRIC_ID,
    RMDT.METRIC_DETAIL_ID,

    CASE
        WHEN RMDT.METRICS_DISPLAY IS NULL
            THEN RMPM.METRICS_DISP
        ELSE RMDT.METRICS_DISPLAY
    END AS METRIC_NAME,

    RMDT.METRIC_NAME_CASCADE,
    RMDT.THRESHOLD_CASCADE,
    RMDT.METRIC_APPLICABILITY,

    -- PARENT / CHILD INFO
    MCL.PARENT_RMM_ID,
    MCL.CHILD_RMM_ID,

    -- REMOVE ACTION INFO
    MCA.CASCADE_DT_STR AS CASCADE_DT,

    COALESCE(
        UT.USER_NAME,
        (
            SELECT UT2.USER_NAME
            FROM USER_TAB UT2
            WHERE UT2.PSID = MCA.ACTION_BY
            FETCH FIRST 1 ROW ONLY
        )
    ) AS ACTION_BY,

    -- VERSIONING
    MCL.CASCADE_VERSION,
    MCL.CASCADE_STATUS,
    MCL.IS_ACTIVE,

    MCA.IS_ACTIVE AS MCA_IS_ACTIVE,

    -- PENDING CHILDREN
    CASE
        WHEN EXISTS (
            SELECT 1
            FROM METRIC_CASCADE_ACTIONS MCA2
            WHERE MCA2.METRIC_DETAIL_ID =
                  RMDT.METRIC_DETAIL_ID
              AND MCA2.PARENT_RMM_ID =
                  MCL.PARENT_RMM_ID
              AND MCA2.IS_ACTIVE = 'Y'
              AND MCA2.STATUS IN (
                    'Awaiting Acceptance',
                    'In Progress'
              )
        )
        THEN 'Y'
        ELSE 'N'
    END AS HAS_PENDING_CHILDREN,

    -- LOCAL / GLOBAL
    CASE
        WHEN EXISTS (
            SELECT 1
            FROM METRIC_CASCADE_ACTIONS MCA3
            WHERE MCA3.TARGET_METRIC_DETAIL_ID =
                  RMDT.METRIC_DETAIL_ID
              AND MCA3.STATUS = 'Accepted'
        )
        THEN 'GLOBAL'
        ELSE 'LOCAL'
    END AS METRIC_SOURCE_TYPE,

    -- REMOVE SOURCE TYPE
    CASE
        WHEN MCL.PARENT_RMM_ID = :2
        THEN 'CURRENT_LOCATION'

        WHEN MCL.CHILD_RMM_ID = :2
        THEN 'PARENT_LOCATION'

        ELSE NULL
    END AS REMOVE_SOURCE_TYPE

FROM MEET_INSTC MI

JOIN RAP
    ON MI.MEET_INSTC_ID =
       RAP.RAP_INSTANCE_ID

JOIN RAP_METRICS_DETAILS RMDT
    ON RAP.RAP_ID = RMDT.RAP_ID

JOIN RAP_METRICS_PACK_MAPPING RMPM
    ON RMDT.RAP_METRICS_MAPPING_ID =
       RMPM.RAP_METRICS_MAPPING_ID

-- REMOVE LOCATION TABLE
JOIN (
    SELECT
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION,
        MAX(CASCADE_STATUS) AS CASCADE_STATUS,
        MAX(IS_ACTIVE) AS IS_ACTIVE

    FROM METRIC_CASCADE_LOCATIONS

    WHERE ACTION_TYPE = 'REMOVE'

    GROUP BY
        METRIC_DETAIL_ID,
        PARENT_RMM_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION
) MCL
    ON MCL.METRIC_DETAIL_ID =
       RMDT.METRIC_DETAIL_ID

-- REMOVE ACTION HISTORY
LEFT JOIN (
    SELECT
        METRIC_DETAIL_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION,

        MAX(ACTION_DT) AS CASCADE_DT,

        MAX(ACTION_BY) KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) AS ACTION_BY,

        MAX(ACTION_DT_STR) KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) AS CASCADE_DT_STR,

        MAX(IS_ACTIVE) KEEP (
            DENSE_RANK LAST
            ORDER BY ACTION_DT
        ) AS IS_ACTIVE

    FROM METRIC_CASCADE_ACTIONS

    WHERE ACTION_TYPE = 'REMOVE'

    GROUP BY
        METRIC_DETAIL_ID,
        CHILD_RMM_ID,
        CASCADE_VERSION
) MCA
    ON MCA.METRIC_DETAIL_ID =
       MCL.METRIC_DETAIL_ID
   AND MCA.CASCADE_VERSION =
       MCL.CASCADE_VERSION
   AND MCA.CHILD_RMM_ID =
       MCL.CHILD_RMM_ID

LEFT JOIN USER_TAB UT
    ON UT.EMP_ID = MCA.ACTION_BY

WHERE (
        MCL.PARENT_RMM_ID = :2
        OR
        MCL.CHILD_RMM_ID = :2
      )
  AND MI.MEET_INSTC_ID = :3
  AND RMDT.METRIC_FLAG = 'removed'
