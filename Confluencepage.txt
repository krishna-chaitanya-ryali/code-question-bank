dispense_rows = sql_getData(
    396,
    [
        cascaded_metric_detail_id
    ]
)


if review_rows and review_rows != -1:

    for row in review_rows:

        review_type = (
            row.get("REVIEW_TYPE")
        )

        field_key = (
            row.get("FIELD_KEY")
        )

        if review_type == "Accepted":

            accepted_keys.append(
                field_key
            )


if dispense_rows and dispense_rows != -1:

    for row in dispense_rows:

        field_key = (
            row.get("FIELD_KEY")
        )

        if (
            field_key
            and field_key not in rejected_keys
        ):

            rejected_keys.append(
                field_key
            )
