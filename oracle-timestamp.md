SELECT MASTER_METRIC_NAME, COUNT(*) AS cnt
FROM RAP_MASTER_METRIC_DETAILS
GROUP BY MASTER_METRIC_NAME
HAVING COUNT(*) > 1;

SELECT *
FROM RAP_MASTER_METRIC_DETAILS
WHERE MASTER_METRIC_NAME IN (
    SELECT MASTER_METRIC_NAME
    FROM RAP_MASTER_METRIC_DETAILS
    GROUP BY MASTER_METRIC_NAME
    HAVING COUNT(*) > 1
)
ORDER BY MASTER_METRIC_NAME;
